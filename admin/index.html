// News At Noire — GitHub OAuth broker for Decap CMS (Cloudflare Worker) // Paste this WHOLE file into the worker's "Edit code" editor, then Deploy. // Requires two Worker Secrets: GITHUB_CLIENT_ID and GITHUB_CLIENT_SECRET // (Worker > Settings > Variables and Secrets). const OAUTH_AUTHORIZE = "https://github.com/login/oauth/authorize"; const OAUTH_TOKEN = "https://github.com/login/oauth/access_token"; function renderResult(status, content) { // Decap listens for a postMessage from this popup window. return ` `; } export default { async fetch(request, env) { const url = new URL(request.url); const { pathname } = url; // Step 1: /auth -> redirect the user to GitHub's authorize screen if (pathname === "/auth") { const redirectUri = `${url.origin}/callback`; const authUrl = new URL(OAUTH_AUTHORIZE); authUrl.searchParams.set("client_id", env.GITHUB_CLIENT_ID); authUrl.searchParams.set("redirect_uri", redirectUri); authUrl.searchParams.set("scope", url.searchParams.get("scope") || "repo,user"); authUrl.searchParams.set("state", crypto.randomUUID()); return Response.redirect(authUrl.toString(), 302); } // Step 2: /callback -> exchange the code for a token, hand it to Decap if (pathname === "/callback") { const code = url.searchParams.get("code"); if (!code) { return new Response("Missing code", { status: 400 }); } const tokenRes = await fetch(OAUTH_TOKEN, { method: "POST", headers: { "Content-Type": "application/json", "Accept": "application/json", "User-Agent": "newsatnoire-oauth", }, body: JSON.stringify({ client_id: env.GITHUB_CLIENT_ID, client_secret: env.GITHUB_CLIENT_SECRET, code, }), }); const data = await tokenRes.json(); if (data.error || !data.access_token) { return new Response( renderResult("error", { message: data.error || "no token" }), { headers: { "Content-Type": "text/html" } } ); } return new Response( renderResult("success", { token: data.access_token, provider: "github" }), { headers: { "Content-Type": "text/html" } } ); } // Health check return new Response("News At Noire OAuth broker is running. Use /auth to log in.", { headers: { "Content-Type": "text/plain" }, }); }, };
